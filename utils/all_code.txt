export function getCategorySum(type, transactions, categories, year, monthIndex) {
  const filtered = transactions.filter((t) => {
    const d = new Date(t.date);

    if (d.getMonth() !== monthIndex || d.getFullYear() !== year) return false;

    const cat = categories.find(
      (c) => c.id === (t.categoryId || t.category_id)
    );

    return cat?.type === type;
  });

  return filtered.reduce((sum, t) => sum + t.amount, 0);
}

  export function formatAmount(value) {

  if (Math.abs(rounded) < 0.005) return "0.00";

  return rounded.toFixed(2);
}

export function getSuggestions(
  categoryId,
  inputValue,
  transactions,
  suggestionBlacklist
) {

    if (!inputValue || inputValue.length < 2) return [];

    const lowerInput = inputValue.toLowerCase();

    const notes = transactions
      .filter(
        (t) =>
          t.categoryId === categoryId &&
          t.note &&
          t.note.trim() !== ""
      )
      .map((t) => t.note.trim());

    const frequency = {};

    notes.forEach((note) => {
      const key = note.toLowerCase();

      if (!frequency[key]) {
        frequency[key] = {
          original: note,
          count: 0
        };
      }

      frequency[key].count += 1;
    });

    const sorted = Object.values(frequency)
      .sort((a, b) => b.count - a.count)
      .map((item) => item.original);

    return sorted
      .filter((note) =>
        note.toLowerCase().startsWith(lowerInput)
      )
      .filter((note) =>
        !suggestionBlacklist[categoryId]?.includes(note)
      );
  };

  export function getTopCategories(
  type,
  transactions,
  categories,
  monthKey
) {

    // 1️⃣ tylko bieżący miesiąc + typ
    const filtered = transactions.filter((t) => {
      if (t.date !== monthKey) return false;

      const cat = categories.find(
  (c) => c.id === (t.categoryId || t.category_id)
);
      return cat?.type === type;
    });

    // 2️⃣ liczenie użyć
    const counts = {};

    filtered.forEach((t) => {
      counts[t.categoryId] = (counts[t.categoryId] || 0) + 1;
    });

    // 3️⃣ zamiana na tablicę + sprawdzenie czy to najniższy poziom
    const result = Object.entries(counts)
      .map(([categoryId, count]) => {
        const category = categories.find((c) => c.id === categoryId);
        if (!category) return null;

        const hasChildren = categories.some(
          (c) => c.parentId === category.id
        );

        return {
          category,
          count,
          hasChildren,
        };
      })
      .filter(Boolean)
      .filter((item) => !item.hasChildren); // tylko najniższy poziom

    // 4️⃣ sortowanie malejąco
    result.sort((a, b) => b.count - a.count);

    // 5️⃣ max 3
    return result.slice(0, 3);
  };

export function getCategorySum(categoryIdOrType, transactions, categories, monthKey) {
  const filtered = transactions.filter((t) => {
    const currentId = t.categoryId || t.category_id;
    const category = categories.find((c) => c.id === currentId);
    if (!category) return false;

    if (monthKey) {
      const [year, month] = monthKey.split("-").map(Number);
      const d = new Date(t.date);

      if (d.getFullYear() !== year || d.getMonth() !== month - 1) {
        return false;
      }
    }

    if (categoryIdOrType === "income" || categoryIdOrType === "expense") {
      return category.type === categoryIdOrType;
    }

    if (currentId === categoryIdOrType) return true;

    const isChild = categories.some(
      (c) => c.id === currentId && c.parentId === categoryIdOrType
    );

    const isGrandChild = categories.some((c) => {
      const parent = categories.find((p) => p.id === c.parentId);
      return c.id === currentId && parent?.parentId === categoryIdOrType;
    });

    return isChild || isGrandChild;
  });

  return filtered.reduce((sum, t) => sum + t.amount, 0);
}
